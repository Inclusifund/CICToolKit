    <script type="text/babel">

        const { useState, useEffect } = React;

        const { Search, Filter, X, ExternalLink, ChevronDown, ArrowLeft } = lucide;

        // Convex Client

        const CONVEX_URL = "https://terrific-bloodhound-927.convex.cloud";

        const convexClient = new Convex.ConvexHttpClient(CONVEX_URL);

        const App = () => {

            const [grants, setGrants] = useState([]);

            const [filteredGrants, setFilteredGrants] = useState([]);

            const [loading, setLoading] = useState(true);

            const [searchTerm, setSearchTerm] = useState('');

            const [selectedSectors, setSelectedSectors] = useState([]);

            const [selectedRegions, setSelectedRegions] = useState([]);

            const [selectedStatus, setSelectedStatus] = useState('Open');

            const [sortBy, setSortBy] = useState('name');

            const [showFilters, setShowFilters] = useState(false);

            // Fetch grants from Convex

            useEffect(() => {

                const fetchGrants = async () => {

                    try {

                        const result = await convexClient.query("grants:getAllGrants");

                        // Transform data to match UI expectations

                        const transformed = result.map(g => ({

                            ...g,

                            name: g.grantName || 'Unnamed Grant',

                            minAmount: g.amountMin || 0,

                            maxAmount: g.amountMax || 0,

                            sectors: g.eligibleSectors || [],

                            regions: [g.geographicFocus || 'UK-wide'],

                            status: g.status === 'Open Now' ? 'Open' : g.status

                        }));

                        setGrants(transformed);

                        setFilteredGrants(transformed);

                    } catch (error) {

                        console.error("Error fetching grants:", error);

                    } finally {

                        setLoading(false);

                    }

                };

                fetchGrants();

            }, []);

            // Extract unique sectors and regions

            const allSectors = [...new Set(grants.flatMap(g => g.sectors))].sort();

            const allRegions = [...new Set(grants.flatMap(g => g.regions))].sort();

            // Apply filters

            useEffect(() => {

                let filtered = grants;

                // Status filter

                if (selectedStatus) {

                    filtered = filtered.filter(g => g.status === selectedStatus);

                }

                // Search filter

                if (searchTerm) {

                    filtered = filtered.filter(g =>

                        g.name.toLowerCase().includes(searchTerm.toLowerCase()) ||

                        g.funder.toLowerCase().includes(searchTerm.toLowerCase()) ||

                        g.description.toLowerCase().includes(searchTerm.toLowerCase())

                    );

                }

                // Sector filter

                if (selectedSectors.length > 0) {

                    filtered = filtered.filter(g =>

                        g.sectors.some(s => selectedSectors.includes(s))

                    );

                }

                // Region filter

                if (selectedRegions.length > 0) {

                    filtered = filtered.filter(g =>

                        g.regions.includes('UK-wide') ||

                        g.regions.some(r => selectedRegions.includes(r))

                    );

                }

                // Sort

                filtered = [...filtered].sort((a, b) => {

                    if (sortBy === 'name') return a.name.localeCompare(b.name);

                    if (sortBy === 'maxAmount') return b.maxAmount - a.maxAmount;

                    if (sortBy === 'minAmount') return a.minAmount - b.minAmount;

                    if (sortBy === 'difficulty') return a.applicationDifficulty - b.applicationDifficulty;

                    return 0;

                });

                setFilteredGrants(filtered);

            }, [grants, searchTerm, selectedSectors, selectedRegions, selectedStatus, sortBy]);

            const toggleSector = (sector) => {

                setSelectedSectors(prev =>

                    prev.includes(sector) ? prev.filter(s => s !== sector) : [...prev, sector]

                );

            };

            const toggleRegion = (region) => {

                setSelectedRegions(prev =>

                    prev.includes(region) ? prev.filter(r => r !== region) : [...prev, region]

                );

            };

            const clearFilters = () => {

                setSearchTerm('');

                setSelectedSectors([]);

                setSelectedRegions([]);

                setSelectedStatus('Open');

                setSortBy('name');

            };

            if (loading) {

                return (

                    <div className="min-h-screen bg-brand-dark flex items-center justify-center">

                        <div className="text-center">

                            <div className="inline-block animate-spin rounded-full h-16 w-16 border-b-2 border-brand-primary mb-4"></div>

                            <p className="text-brand-slate text-lg">Loading grants...</p>

                        </div>

                    </div>

                );

            }

            return (

                <div className="min-h-screen bg-brand-dark text-brand-light font-sans">

                    {/* Header */}

                    <header className="bg-brand-card border-b border-brand-border sticky top-0 z-50">

                        <div className="container mx-auto px-4 py-4">

                            <div className="flex items-center justify-between">

                                <div className="flex items-center gap-4">

                                    <a href="grants-onboarding.html" className="text-brand-slate hover:text-brand-primary transition-colors">

                                        <ArrowLeft size={24} />

                                    </a>

                                    <h1 className="font-serif text-2xl md:text-3xl text-brand-primary font-bold">

                                        InclusiFund Grants

                                    </h1>

                                </div>

                                    onClick={() => setShowFilters(!showFilters)}

                                    className="md:hidden bg-brand-primary/10 text-brand-primary px-4 py-2 rounded-lg flex items-center gap-2"

                                >

                                    <Filter size={18} />

                                    Filters

                                </button>

                            </div>

                        </div>

                    </header>

                    <div className="container mx-auto px-4 py-8">

                        <div className="flex flex-col md:flex-row gap-8">

                                    <div className="flex items-center justify-between mb-6">

                                        <h2 className="font-serif text-xl text-brand-white">Filters</h2>

                                        <button

                                            onClick={clearFilters}

                                            className="text-brand-slate hover:text-brand-primary text-sm transition-colors"

                                        >

                                            Clear All

                                        </button>

                                    </div>

                                    {/* Search */}

                                    <div className="mb-6">

                                            <input

                                                type="text"

                                                value={searchTerm}

                                                onChange={(e) => setSearchTerm(e.target.value)}

                                                placeholder="Grant name or funder..."

                                                className="w-full bg-brand-dark border border-brand-border rounded-lg pl-10 pr-4 py-2 text-brand-white placeholder-brand-slate focus:outline-none focus:border-brand-primary"

                                            />

                                        </div>

                                    </div>

                                    {/* Status */}

                                    <div className="mb-6">

                                        <label className="block text-brand-slate text-sm mb-2">Status</label>

                                        <select

                                            value={selectedStatus}

                                            onChange={(e) => setSelectedStatus(e.target.value)}

                                            className="w-full bg-brand-dark border border-brand-border rounded-lg px-4 py-2 text-brand-white focus:outline-none focus:border-brand-primary"

                                        >

                                            <option value="">All</option>

                                            <option value="Open">Open</option>

                                        </select>

                                    </div>

                                    {/* Sort By */}

                                    <div className="mb-6">

                                        <label className="block text-brand-slate text-sm mb-2">Sort By</label>

                                        <select

                                            value={sortBy}

                                            onChange={(e) => setSortBy(e.target.value)}

                                            className="w-full bg-brand-dark border border-brand-border rounded-lg px-4 py-2 text-brand-white focus:outline-none focus:border-brand-primary"

                                        >

                                            <option value="name">Name (A-Z)</option>

                                            <option value="maxAmount">Highest Amount</option>

                                            <option value="minAmount">Lowest Amount</option>

                                            <option value="difficulty">Easiest First</option>

                                        </select>

                                    </div>

                                    {/* Sectors */}

                                    <div className="mb-6">

                                        <label className="block text-brand-slate text-sm mb-2">Sectors</label>

                                        <div className="space-y-2 max-h-48 overflow-y-auto custom-scrollbar">

                                            {allSectors.map(sector => (

                                                <label key={sector} className="flex items-center gap-2 cursor-pointer hover:bg-brand-dark/50 p-2 rounded">

                                                    <input

                                                        type="checkbox"

                                                        checked={selectedSectors.includes(sector)}

                                                        onChange={() => toggleSector(sector)}

                                                        className="w-4 h-4 accent-brand-primary"

                                                    />

                                                    <span className="text-sm text-brand-light">{sector}</span>

                                                </label>

                                            ))}

                                    {/* Regions */}

                                    <div className="mb-6">

                                        <label className="block text-brand-slate text-sm mb-2">Regions</label>

                                        <div className="space-y-2 max-h-48 overflow-y-auto custom-scrollbar">

                                            {allRegions.map(region => (

                                                <label key={region} className="flex items-center gap-2 cursor-pointer hover:bg-brand-dark/50 p-2 rounded">

                                                    <input

                                                        type="checkbox"

                                                        checked={selectedRegions.includes(region)}

                                                        onChange={() => toggleRegion(region)}

                                                        className="w-4 h-4 accent-brand-primary"

                                                    />

                                                    <span className="text-sm text-brand-light">{region}</span>

                                                </label>

                                            ))}

                                        </div>

                                </div>

                            </aside>

                            {/* Main Content */}

                            <main className="flex-1">

                                <div className="mb-6">

                                        {filteredGrants.length} {filteredGrants.length === 1 ? 'Grant' : 'Grants'} Found

                                    </h2>

                                    {(selectedSectors.length > 0 || selectedRegions.length > 0 || searchTerm) && (

                                        <div className="flex flex-wrap gap-2 mt-4">

                                            {selectedSectors.map(sector => (

                                                    {sector}

                                            {selectedRegions.map(region => (
